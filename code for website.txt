    <script>
        const modal = document.getElementById('modalOverlay');
        const giftInput = document.getElementById('gift_requested');
        const videoSection = document.getElementById('embedded-video-section');

        function openModal(itemType) {
            giftInput.value = itemType; 
            modal.style.display = 'flex'; 
        }

        function closeModal() {
            modal.style.display = 'none';
        }

        window.onclick = function(event) {
            if (event.target == modal) {
                closeModal();
            }
        }

        function handleFormSubmit(e) {
            e.preventDefault(); 
            const form = e.target;
            const data = new FormData(form);
            const giftType = data.get('gift_requested'); 
            
            fetch(form.action, {
                method: form.method,
                body: data,
                headers: {
                    'Accept': 'application/json'
                }
            }).then(response => {
                if (response.ok) {
                    modal.style.display = 'none'; 
                    
                    if (giftType === 'video') {
                        // REVEAL THE EMBEDDED VIDEO SECTION
                        videoSection.style.display = 'block';
                        // Scroll to the video
                        videoSection.scrollIntoView({ behavior: 'smooth' });
                    } else {
                        alert('تم التسجيل بنجاح! سيتم تحويلك للكتاب...');
                        // UPDATED BOOK LINK
                        window.location.href = "https://drive.google.com/file/d/18Xykhhu52FDVLY376TU54FdvY6Txgcrs/view?usp=drive_link";
                    }
                } else {
                    alert('حدث خطأ ما. يرجى المحاولة مرة أخرى.');
                }
            }).catch(error => {
                alert('حدث خطأ في الاتصال. يرجى المحاولة مرة أخرى.');
            });
        }
    </script>
